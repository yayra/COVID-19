-- Написать примерный DDL для каждого из таблиц
-- TASK1. DDL in PostgreSQL
-- USING DDL statement CREATE TABLE to create all the tables in the given scheme. 

CREATE TABLE productLine (
    productLine int NOT NULL PRIMARY KEY,
    textDescription VARCHAR(250),
	htmlDescription VARCHAR(250),
    image BOOLEAN
);

CREATE TABLE products (
   productCode int NOT NULL PRIMARY KEY,
   productName VARCHAR(50) NOT NULL, 
   productLine int NOT NULL,
   prodcutScale int NOT NULL, 
   productVendor VARCHAR (50),
   productDescription VARCHAR(250),
	quantityInStock int,
	buyPrice VARCHAR(50),
	MSRP VARCHAR(50)
	);
	
CREATE TABLE orderdetails(
   orderNumber int NOT NULL PRIMARY KEY,
   productCode int NOT NULL,
   quantityOrdered int,
   priceEach  varchar(50),
   orderLineNumber int
);

CREATE TABLE employees(
    employeeNumber int NOT NULL PRIMARY KEY,
    lastName VARCHAR (50) NOT NULL, 
    firstName VARCHAR (50) NOT NULL,
    extention int,
    email VARCHAR (50),
    officeCode VARCHAR (50),
    reportsTo VARCHAR (50) NOT NULL, 
    jobTItle VARCHAR (50) NOT NULL
);
	
CREATE TABLE offices(
    officeCode int NOT NULL PRIMARY KEY,
    city VARCHAR (50) NOT NULL,
    phone VARCHAR (50) NOT NULL,
    addressLine1 VARCHAR(50) NOT NULL,
    addressLine2 VARCHAR(50),
	statei VARCHAR(50) NOT NULL, 
	country VARCHAR(50) NOT NULL,
	postalCode int NOT NULL,
	territory VARCHAR(50)
	);

CREATE TABLE customers(
    customerNumber int NOT NULL PRIMARY KEY,
    CustomerName Varchar(50) NOT NULL,
    contactLastName Varchar(50) NOT NULL,
    contactFirstName Varchar(50) NOT NULL,
    phone int NOT NULL,
	city VARCHAR (50) NOT NULL,
	statei VARCHAR(50) NOT NULL, 
	postalCode int NOT NULL,
	salesRepEmployeeNumber Int NOT NULL,
	creditLimit VARCHAR(50)
	);
	
CREATE TABLE orders(
	orderNumber int NOT NULL PRIMARY KEY,
	orderDate DATE NOT NULL,
	requiredDate DATE,
	shippedDate DATE,
	status Boolean,
	commen VARCHAR(5),
	customerNumber  int NOT NULL
	);

CREATE TABLE payments(	
	customerNumber int NOT NULL PRIMARY KEY,
	checkNumber VARCHAR(5) NOT NULL,
	paymentDate DATE, 
	amount int
	);



--USING DDL statements TRUNCATE, DROP, ALTER

TRUNCATE TABLE productLine;

DROP TABLE employees;

ALTER COLUMN
requiredDate 
SET NOT NULL;
